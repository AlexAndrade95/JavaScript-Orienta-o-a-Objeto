Objetos constantes 
Object.freeze

// pessoa -> 123 -> {...} //endereço de memória
const pessoa = { nome: 'Joao' }
pessoa.nome = 'Pedro'
console.log(pessoa)
// { nome: 'Pedro' }

// pessoa -> 456 -> {...} //endereço de memória
// pessoa = { nome: 'Ana' }

Object.freeze(pessoa) // congelar o objeto (não conseguirá mexer mais no objeto)

pessoa.nome = 'Maria'
pessoa.end = 'Rua ABC'
delete pessoa.nome

console.log(pessoa.nome)
// { nome: 'Pedro' }
console.log(pessoa)
// Pedro
  { nome: 'Pedro' }

const pessoaConstante = Object.freeze({ nome: 'Joao' }) // para criar um objeto constante
pessoaConstante.nome = 'Maria'
console.log(pessoaConstante)
// { nome: 'Joao' }
